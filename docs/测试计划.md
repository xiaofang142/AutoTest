```python
# tests/unit/services/test_analysis_service.py

import pytest
from datetime import datetime, timedelta

from app.services.analysis_service import CrossDimensionAnalyzer
from app.domain.models.run import StepExecutionRecord, Anomaly, EvidenceChain


class TestCausalRuleEngine:
    """综合分析引擎的规则引擎测试"""

    @pytest.fixture
    def analyzer(self):
        return CrossDimensionAnalyzer()

    def test_api_error_to_console_error(self, analyzer):
        """API 500 应该关联到后续的 console Error"""
        api_error = Anomaly(
            dimension="api",
            type="error",
            timestamp=datetime.now(),
            data={"url": "/api/v1/orders", "status": 500}
        )
        console_error = Anomaly(
            dimension="console",
            type="error",
            timestamp=datetime.now() + timedelta(milliseconds=500),
            data={"message": "Uncaught TypeError: Cannot read 'orderId'"}
        )
        
        assert analyzer.rule_engine.is_causally_related(api_error, console_error)

    def test_unrelated_events_not_linked(self, analyzer):
        """不相关的事件不应该被关联"""
        console_warning = Anomaly(
            dimension="console",
            type="warning",
            timestamp=datetime.now(),
            data={"message": "Deprecated API warning"}
        )
        ui_issue = Anomaly(
            dimension="ui",
            type="broken",
            timestamp=datetime.now() + timedelta(seconds=10),
            data={"visible_texts": []}
        )
        
        assert not analyzer.rule_engine.is_causally_related(console_warning, ui_issue)

    def test_reverse_time_order_not_linked(self, analyzer):
        """后发生的事件不能是前面事件的原因"""
        later = Anomaly(dimension="api", type="error", timestamp=datetime.now())
        earlier = Anomaly(dimension="console", type="error",
                         timestamp=datetime.now() - timedelta(seconds=10))
        
        assert not analyzer.rule_engine.is_causally_related(later, earlier)

    def test_too_close_events_not_linked(self, analyzer):
        """50ms 内的两个事件视为并发"""
        api_error = Anomaly(
            dimension="api", type="error", timestamp=datetime.now(),
            data={"url": "/api/test", "status": 500}
        )
        console_error = Anomaly(
            dimension="console", type="error",
            timestamp=datetime.now() + timedelta(milliseconds=20),
            data={"message": "Error"}
        )
        
        assert not analyzer.rule_engine.is_causally_related(api_error, console_error)

    def test_api_cascade_401(self, analyzer):
        """Token 过期导致的批量 401"""
        auth_error = Anomaly(
            dimension="api", type="error",
            timestamp=datetime.now(),
            data={"url": "/api/v1/auth/refresh", "status": 401}
        )
        order_error = Anomaly(
            dimension="api", type="error",
            timestamp=datetime.now() + timedelta(milliseconds=100),
            data={"url": "/api/v1/orders", "status": 401}
        )
        
        assert analyzer.rule_engine.is_causally_related(auth_error, order_error)
```

## 5. 性能测试

### 5.1 性能目标

```yaml
API 性能目标:
  常规查询 (项目列表、执行历史):
    P50: < 100ms
    P95: < 200ms
    P99: < 500ms
  
  复杂查询 (报告生成、缺陷详情):
    P50: < 500ms
    P95: < 2s
    P99: < 5s
  
  AI 调用 (文档解析、根因分析):
    异步任务，不卡 API 响应
    文档解析: < 3min (10 页文档)
    根因分析: < 30s

执行器性能目标:
  单步执行（Web）:
    简单操作（点击/输入）: < 3s
    复杂操作（多步 AI 定位）: < 8s
  
  并发:
    单机 4 个浏览器: 步骤延迟增加 < 20%
    8 节点集群: 支持 32 个并发用例
```

### 5.2 负载测试场景

```yaml
测试工具: locust / k6

场景 1: API 常规负载
  并发用户: 50
  持续时间: 10min
  接口: GET /projects, GET /runs, GET /defects
  目标: P95 < 200ms, 错误率 < 0.1%

场景 2: API 混合负载
  并发用户: 30
  持续时间: 30min
```

## 附录 B: Flaky Test 管理策略

### B.1 Flaky Test 定义与识别

```yaml
Flaky Test 定义:
  同一测试用例在相同条件下，多次执行结果不一致。
  
识别标准:
  ┌────────────────────────────────┬──────────┐
  │ 标准                            │ 阈值      │
  ├────────────────────────────────┼──────────┤
  │ 同用例 5 次执行通过率 < 100%      │ Flaky    │
  │ 同用例 10 次执行通过率 > 80%      │ Likely Flaky │
  │ 连续 3 次失败后重试通过            │ Flaky    │
  │ 不同平台同一用例结果不一致          │ Investigate │
  └────────────────────────────────┴──────────┘

自动标记:
  - 执行完成后自动分析 Flakyness
  - 标记为 flaky 的用例在报告中高亮
  - 每天发送 Flaky Test 报告

Flaky 原因分类:
  ├── 环境问题 (60%): 网络波动、浏览器版本、资源争用
  ├── AI 定位 (20%): 视觉定位偶发不准
  ├── 时序问题 (10%): 等待时间不足、动画未完成
  └── 被测应用 (10%): 服务偶发不稳定
```

### B.2 自动重试策略

```python
# flaky_retry.py

class FlakyRetryStrategy:
    """Flaky Test 自动重试策略"""
    
    def __init__(self):
        self.max_retries = 3
        self.flaky_threshold = 0.7  # 通过率 < 70% 视为 flaky
    
    async def should_retry(self, step_result: StepExecutionRecord) -> bool:
        """判断是否需要重试"""
        # 只有 failed 和 uncertain 需要重试
        if step_result.status not in ("failed", "uncertain"):
            return False
        
        # UI 校验失败最常见的 flaky 原因，最多重试 2 次
        if step_result.verifications.ui.status == "failed":
            return step_result.retry_count < 2
        
        # API 校验失败通常不是 flaky，重试 1 次验证
        if step_result.verifications.api.status == "failed":
            return step_result.retry_count < 1
        
        return step_result.retry_count < self.max_retries
    
    async def analyze_flaky(self, run_id: str) -> list[FlakyReport]:
        """分析执行中的 flaky test"""
        steps = await step_repo.get_by_run(run_id)
        
        flaky_cases = []
        for step in steps:
            # 查找同用例同步骤的历史执行
            history = await step_repo.get_history(
                case_id=step.case_id,
                step_index=step.step_index,
                limit=10
            )
            
            if len(history) >= 3:
                pass_rate = sum(1 for s in history if s.status == "passed") / len(history)
                if pass_rate < self.flaky_threshold:
                    flaky_cases.append(FlakyReport(
                        case_id=step.case_id,
                        step_index=step.step_index,
                        pass_rate=pass_rate,
                        attempt_count=len(history),
                        recommended_action="review_and_fix"
                    ))
        
        return flaky_cases

# 应用场景：
# 1. 用例失败 → 自动重试（最多 3 次）
# 2. 重试通过 → 标记为 flaky（不合并到最终报告）
# 3. 重试仍失败 → 确认为缺陷
```
